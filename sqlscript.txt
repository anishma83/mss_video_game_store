//Suppliers table
CREATE TABLE
    DB2ADMIN.Suppliers
    (
        SupplierID INTEGER NOT NULL GENERATED ALWAYS AS IDENTITY,
        CompanyName VARCHAR(50) NOT NULL,
        ContactFName VARCHAR(30) NOT NULL,
        ContactLName VARCHAR(40) NOT NULL,
        ContactTitle VARCHAR(30) NOT NULL,
        AddressLine1 VARCHAR(60) NOT NULL,
        AddressLine2 VARCHAR(40),
        City VARCHAR(15) NOT NULL,
        USstate VARCHAR(25) NOT NULL,
        Postal VARCHAR(15) NOT NULL,
        Country VARCHAR(50) NOT NULL,
        Phone VARCHAR(25) NOT NULL,
        Email VARCHAR(75) NOT NULL,
        Website VARCHAR(100) NOT NULL,
        CurrentOrder VARCHAR(1) NOT NULL,
        CustomerID VARCHAR(50) NOT NULL,
        logo VARCHAR(75),
        notes VARCHAR(255),
        PRIMARY KEY (SupplierID),
        CONSTRAINT SupplierID UNIQUE (SupplierID)
    )
	
//Shipper table
CREATE TABLE
    DB2ADMIN.Shippers
    (
        ShipperID INTEGER NOT NULL,
        ShipperName VARCHAR(50) NOT NULL,
        Phone VARCHAR(25),
        PRIMARY KEY (ShipperID),
        CONSTRAINT ShipperID UNIQUE (ShipperID)
    )
	
//Category TABLE
CREATE TABLE
    DB2ADMIN.Category
    (
        CategoryID INTEGER NOT NULL,
        Name VARCHAR(40) NOT NULL,
        Description VARCHAR(255) NOT NULL,
        PRIMARY KEY (CategoryID),
        CONSTRAINT CategoryID UNIQUE (CategoryID)
    )
	
//EcommerceCustomer TABLE
CREATE TABLE
    DB2ADMIN.EcommerceCustomer
    (
        CustomerID INTEGER NOT NULL GENERATED ALWAYS AS IDENTITY,
        FirstName VARCHAR(30) NOT NULL,
        LastName VARCHAR(50) NOT NULL,
        Address1 VARCHAR(60) NOT NULL,
        Address2 VARCHAR(40),
        City VARCHAR(15) NOT NULL,
        UsState VARCHAR(25) NOT NULL,
        Postal VARCHAR(15) NOT NULL,
        Country VARCHAR(50) NOT NULL,
        Phone VARCHAR(25) NOT NULL,
        Email VARCHAR(75) NOT NULL,
        Password VARCHAR(15) NOT NULL,
        CreditCard VARCHAR(19) NOT NULL,
        CreditCardID VARCHAR(1) NOT NULL,
        CreditExMo VARCHAR(2) NOT NULL,
        CeditExYr VARCHAR(2) NOT NULL,
        BillingAddress VARCHAR(60) NOT NULL,
        BillingAddress2 VARCHAR(40),
        BillingCity VARCHAR(25) NOT NULL,
        BillingState VARCHAR(25) NOT NULL,
        BillingPostal VARCHAR(15) NOT NULL,
        BillingCountry VARCHAR(50) NOT NULL,
        DateEntered DATE NOT NULL,
		Member VARCHAR(1),
        PRIMARY KEY (CustomerID),
        CONSTRAINT CustomerID UNIQUE (CustomerID)
    )
	
//Product Table
CREATE TABLE
    DB2ADMIN.Product
    (
        ProductID INTEGER NOT NULL GENERATED ALWAYS AS IDENTITY,
        SKU VARCHAR(50) NOT NULL,
        SupplierProductID VARCHAR(50) NOT NULL,
        ProductName VARCHAR(50) NOT NULL,
        SupplierID INTEGER NOT NULL,
        CategoryID INTEGER NOT NULL,
        UnitPrice DOUBLE NOT NULL,
        UnitsInStock INTEGER NOT NULL,
        UnitesOnOrder INTEGER NOT NULL,
        ReOrdeLevel INTEGER NOT NULL,
        ProductAvailability VARCHAR(1) NOT NULL,
        Image VARCHAR(100),
        notes VARCHAR(255),
        PRIMARY KEY (ProductID),
        CONSTRAINT ProductID UNIQUE (ProductID),
        CONSTRAINT Supplier FOREIGN KEY (SupplierID) REFERENCES DB2ADMIN.SUPPLIERS (SUPPLIERID),
        CONSTRAINT Category FOREIGN KEY (CategoryID) REFERENCES DB2ADMIN.CATEGORY (CATEGORYID)
    )
	
//Orders Table
CREATE TABLE
    DB2ADMIN.Orders
    (
        OrderID INTEGER NOT NULL GENERATED ALWAYS AS IDENTITY,
        CustomerID INTEGER NOT NULL,
        OrderDate DATE NOT NULL,
        ShipDate DATE NOT NULL,
        ShipperID INTEGER NOT NULL,
        Freight DOUBLE NOT NULL,
        SalesTax DOUBLE NOT NULL,
        OrderTimeStamp VARCHAR(50) NOT NULL,
        TransStatus VARCHAR(50),
        ErrLoc VARCHAR(50),
        ErrMsg VARCHAR(50),
        Fulfilled VARCHAR(1) NOT NULL,
        Deleted VARCHAR(1) NOT NULL,
        Paid VARCHAR(1) NOT NULL,
        PaymentDate DATE,
        PRIMARY KEY (OrderID),
        CONSTRAINT OrderID UNIQUE (OrderID),
        CONSTRAINT Customer FOREIGN KEY (CustomerID) REFERENCES DB2ADMIN.ECOMMERCECUSTOMER
        (CUSTOMERID),
        CONSTRAINT Shipper FOREIGN KEY (ShipperID) REFERENCES DB2ADMIN.SHIPPERS (SHIPPERID)
    )
	
//Order Details Table
CREATE TABLE
    DB2ADMIN.OrderDetails
    (
        OrderDetailID INTEGER NOT NULL GENERATED ALWAYS AS IDENTITY,
        OrderID INTEGER NOT NULL,
        ProductID INTEGER NOT NULL,
        Price DOUBLE NOT NULL,
        quantity INTEGER NOT NULL,
        total DOUBLE NOT NULL,
        Fulfilled VARCHAR(1) NOT NULL,
        BillDate DATE NOT NULL,
        ShipDate DATE NOT NULL,
        ShipperID INTEGER NOT NULL,
        Freight DOUBLE NOT NULL,
        SalesTax DOUBLE NOT NULL,
        PRIMARY KEY (OrderDetailID),
        CONSTRAINT OrderDetails UNIQUE (OrderDetailID),
        CONSTRAINT ORDER FOREIGN KEY (OrderID) REFERENCES DB2ADMIN.ORDERS (ORDERID),
        CONSTRAINT Product FOREIGN KEY (ProductID) REFERENCES DB2ADMIN.PRODUCT (PRODUCTID),
        CONSTRAINT Shipper FOREIGN KEY (ShipperID) REFERENCES DB2ADMIN.SHIPPERS (SHIPPERID)
    )
